import OSM from 'ol/source/OSM';//OSM je skraćenica za OpenStreetMap, što je besplatna i otvorena mapa sveta koju koristi OpenLayers kao izvor podataka za prikazivanje mape.
import TileLayer from 'ol/layer/Tile';//TileLayer je sloj koji se koristi za prikazivanje podataka u obliku pločica (tiles) na mapi. Ove pločice se učitavaju sa izvora podataka, kao što je OSM, i prikazuju se na mapi kao sloj.
import {Map, View} from 'ol';//Map je glavni objekat koji predstavlja mapu, dok View predstavlja pogled na mapu, uključujući centar i nivo zumiranja. Ove klase su ključne za kreiranje i konfiguraciju mape u OpenLayers biblioteci.
import {fromLonLat} from 'ol/proj';//fromLonLat je funkcija koja se koristi za konvertovanje geografskih koordinata (longitude i latitude) u koordinatni sistem koji koristi OpenLayers, što je obično Web Mercator (EPSG:3857). Ovo je važno za pravilno pozicioniranje elemenata na mapi, kao što su markeri ili centar pogleda.

import VectorLayer from 'ol/layer/Vector';//VectorLayer je sloj koji se koristi za prikazivanje vektorskih podataka na mapi, kao što su tačke, linije ili poligoni. Ovaj sloj može sadržavati različite geometrijske oblike i stilove, i često se koristi za prikazivanje prilagođenih podataka ili interaktivnih elemenata na mapi.
import VectorSource from 'ol/source/Vector';//VectorSource je izvor podataka koji se koristi za vektorske slojeve (VectorLayer) u OpenLayers biblioteci. Ovaj izvor može sadržavati različite geometrijske oblike, kao što su tačke, linije ili poligoni, i omogućava dodavanje, uklanjanje ili ažuriranje ovih geometrija na mapi. VectorSource je ključan za rad sa vektorskim podacima i interakciju sa njima na mapi.
import GeoJSON from 'ol/format/GeoJSON';//GeoJSON je format za kodiranje različitih geografskih struktura, kao što su tačke, linije i poligoni, u JSON formatu. U OpenLayers biblioteci, GeoJSON se koristi za učitavanje i prikazivanje geografskih podataka na mapi. Ova klasa omogućava konverziju između GeoJSON formata i geometrijskih objekata koje OpenLayers koristi, što olakšava rad sa geografskim podacima i njihovo prikazivanje na mapi.

import {Style, Circle as CircleStyle, Fill, Stroke, Text} from 'ol/style';//Style je klasa koja se koristi za definisanje izgleda vektorskih elemenata na mapi, kao što su tačke, linije i poligoni. CircleStyle se koristi za stilizovanje tačaka kao krugova, Fill se koristi za popunjavanje poligona, Stroke se koristi za crtanje linija i ivica, dok Text se koristi za prikazivanje teksta na mapi. Ove klase omogućavaju prilagođavanje izgleda vektorskih elemenata i stvaranje vizuelno privlačnih mapa.

import Draw from 'ol/interaction/Draw';//Draw je klasa koja se koristi za omogućavanje interakcije crtanja na mapi. Ova interakcija omogućava korisnicima da crtaju različite geometrijske oblike, kao što su tačke, linije i poligoni, direktno na mapi. Draw interakcija se može konfigurirati da podržava različite vrste geometrija i stilove crtanja, što je korisno za aplikacije koje zahtevaju prilagođeno kreiranje geografskih podataka od strane korisnika.

import MousePosition from 'ol/control/MousePosition';//MousePosition je klasa koja se koristi za prikazivanje trenutne pozicije miša na mapi. Ova kontrola prikazuje koordinate miša u određenom formatu, kao što su geografske koordinate (longitude i latitude) ili koordinate u projekciji mape. MousePosition je korisna za aplikacije koje zahtevaju precizno lociranje ili interakciju sa geografskim podacima na mapi, jer omogućava korisnicima da vide tačne koordinate gde se nalaze na mapi.
import {defaults as defaultControls, ScaleLine} from 'ol/control';//defaultControls je funkcija koja se koristi za dodavanje podrazumevanih kontrola na mapu, kao što su zoom kontrole, navigacione kontrole i druge osnovne funkcionalnosti. ScaleLine je klasa koja se koristi za prikazivanje linije skale na mapi, koja pokazuje odnos između udaljenosti na mapi i stvarne udaljenosti na terenu. Ove kontrole su korisne za poboljšanje interaktivnosti i informativnosti mape, pružajući korisnicima dodatne informacije i funkcionalnosti prilikom rada sa mapom.
import {createStringXY} from 'ol/coordinate';//createStringXY je funkcija koja se koristi za formatiranje koordinata u čitljiv string format. Ova funkcija omogućava prilagođavanje načina na koji se koordinate prikazuju, kao što su decimalne tačke, razmak između koordinata ili drugi formati. createStringXY je korisna za aplikacije koje zahtevaju precizno prikazivanje koordinata, kao što su aplikacije za navigaciju ili geolokacijske usluge, jer omogućava korisnicima da lako razumeju i interpretiraju koordinate koje se prikazuju na mapi.

import GeoTIFF from 'ol/source/GeoTIFF';//GeoTIFF je klasa koja se koristi za učitavanje i prikazivanje GeoTIFF fajlova na mapi. GeoTIFF je format za georeferencirane raster slike, koji sadrži informacije o koordinatnom sistemu i geometriji slike. Ova klasa omogućava rad sa GeoTIFF fajlovima, uključujući učitavanje, prikazivanje i manipulaciju ovih fajlova na mapi, što je korisno za aplikacije koje zahtevaju rad sa georeferenciranim raster podacima.

const turizamLayer = new VectorLayer({//kreiranje vektorskog sloja koji će prikazivati turističke lokacije na mapi
  source: new VectorSource({//izvor podataka za vektorski sloj, koji se kreira pomoću VectorSource klase
    features: new GeoJSON().readFeatures(geojsonObject, {//učitavanje GeoJSON objekta i konvertovanje njegovih funkcija u OpenLayers format pomoću GeoJSON klase
      featureProjection: 'EPSG:3857'//definisanje projekcije u kojoj će se funkcije prikazivati na mapi, što je obično Web Mercator (EPSG:3857) za OpenLayers. Ovo osigurava da se funkcije pravilno pozicioniraju na mapi i da se koordinatni sistemi usklade.
    })
  }),
  style: function(feature) {//  stilizovanje funkcija na vektorskom sloju, gde se za svaku funkciju kreira novi stil koji uključuje krug kao marker i tekst sa imenom lokacije. Boja markera se uzima iz svojstva 'marker-color' funkcije, a tekst se prikazuje iz svojstva 'name'.
    return new Style({//kreiranje novog stila za svaku funkciju na vektorskom sloju
      image: new CircleStyle({//definisanje stila za tačke na mapi, gde se koristi CircleStyle klasa da se prikažu kao krugovi
        radius: 8,//postavljanje radijusa kruga na 8 piksela
        fill: new Fill({color: feature.get('marker-color') || '#1e3e9f'}),//postavljanje boje popunjavanja kruga na vrednost svojstva 'marker-color' funkcije, ili na podrazumevanu boju '#1e3e9f' ako to svojstvo nije definisano
        stroke: new Stroke({color: '#fff', width: 2})//postavljanje boje ivice kruga na belu i širine na 2 piksela
      }),
      text: new Text({
        text: feature.get('name'),
        offsetY: -15,//pomera tekst 15 piksela iznad markera
        fill: new Fill({color: '#000'}),
        stroke: new Stroke({color: '#fff', width: 2})
      })
    });
  }
});
